Plot Creation

# make a top 5
top5 <- outer_space_objects |> 
  filter(!Entity  %in% c('World', 'European Space Agency')) |> 
  group_by(Entity) |> 
  summarise(total = sum(num_objects)) |> 
  arrange(-total) |> 
  pull(Entity) |> head(5)

finalorder <- outer_space_objects |> 
  filter(!Entity  %in% c('World', 'European Space Agency'),
         Entity %in% top5,
         Year == max(Year)) |> 
  arrange(-num_objects) |> 
  pull(Entity) |> 
  factor()
